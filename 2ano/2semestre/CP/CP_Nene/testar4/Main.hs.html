  225 -- Instância de Arbitrary para arestas
  226 instance Arbitrary v => Arbitrary (Edge v) where
  227     arbitrary = do s <- arbitrary
  228                    t <- arbitrary
  229                    return $ Edge {source = s, target = t}
  230 
  231 instance (Ord v, Arbitrary v) => Arbitrary (Graph v) where
  232     arbitrary = aux `suchThat` isValid
  233         where aux = do ns <- arbitrary
  234                        es <- arbitrary 
  235                        return $ Graph {nodes = fromList ns, edges = fromList es}

  246 prop_valid :: Graph Int -> Property
  247 prop_valid g = collect (length (edges g)) $ isValid g
  248 
  249 -- Gerador de DAGs
  250 dag :: (Ord v, Arbitrary v) => Gen (DAG v)
  251 dag = arbitrary `suchThat` isDAG
  252 
  253 prop_dag :: Property
  254 prop_dag = forAll (dag :: Gen (DAG Int)) $ \g -> collect (length (edges g)) $ isDAG g
  255 
  256 -- Gerador de florestas
  257 forest :: (Ord v, Arbitrary v) => Gen (Forest v)
  258 forest = arbitrary `suchThat` isForest
  259 
  260 prop_forest :: Property
  261 prop_forest = forAll (forest :: Gen (Forest Int)) $ \g -> collect (length (edges g)) $ isForest g

  270 -- Exemplo de uma propriedade QuickCheck para testar a função adj          
  271 prop_adj :: Graph Int -> Property
  272 prop_adj g = forAll (elements $ elems $ nodes g) $ \v -> adj g v `isSubsetOf` edges g
  273 
  274 prop_swap :: Graph Int -> Property
  275 prop_swap g = forAll (elements $ elems $ edges g) $ \e -> swap(swap e) `member` edges g
