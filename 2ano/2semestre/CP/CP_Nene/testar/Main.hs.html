  198 -- A instância de Arbitrary para grafos definida abaixo gera grafos
  199 -- com muito poucas arestas, como se pode constatar testando a
  200 -- propriedade prop_valid.
  201 -- Defina uma instância de Arbitrary menos enviesada.
  202 -- Este problema ainda é mais grave nos geradores dag e forest que
  203 -- têm como objectivo gerar, respectivamente, grafos que satisfazem
  204 -- os predicados isDag e isForest. Estes geradores serão necessários
  205 -- para testar propriedades sobre estas classes de grafos.
  206 -- Melhore a implementação destes geradores por forma a serem menos enviesados.
  207 --
  208 
  209 -- Instância de Arbitrary para arestas
  210 instance Arbitrary v => Arbitrary (Edge v) where
  211     arbitrary = do s <- arbitrary
  212                    t <- arbitrary
  213                    return $ Edge {source = s, target = t}
  214 
  215 instance (Ord v, Arbitrary v) => Arbitrary (Graph v) where
  216     arbitrary = aux `suchThat` isValid
  217         where aux = do ns <- arbitrary
  218                        es <- arbitrary
  219                        return $ Graph {nodes = fromList ns, edges = fromList es}
  220  
  221 prop_valid :: Graph Int -> Property
  222 prop_valid g = collect (length (edges g)) $ isValid g
  223 
  224 -- Gerador de DAGs
  225 dag :: (Ord v, Arbitrary v) => Gen (DAG v)
  226 dag = arbitrary `suchThat` isDAG
  227 
  228 prop_dag :: Property
  229 prop_dag = forAll (dag :: Gen (DAG Int)) $ \g -> collect (length (edges g)) $ isDAG g
  230 
  231 -- Gerador de florestas
  232 forest :: (Ord v, Arbitrary v) => Gen (Forest v)
  233 forest = arbitrary `suchThat` isForest
  234 
  235 prop_forest :: Property
  236 prop_forest = forAll (forest :: Gen (Forest Int)) $ \g -> collect (length (edges g)) $ isForest g
