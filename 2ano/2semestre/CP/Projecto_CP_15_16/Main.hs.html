  212 -- Instância de Arbitrary para arestas
  213 instance Arbitrary v => Arbitrary (Edge v) where
  214     arbitrary = do s <- arbitrary
  215                    t <- arbitrary
  216                    return $ Edge {source = s, target = t}
  217 
  218 instance (Ord v, Arbitrary v) => Arbitrary (Graph v) where
  219     arbitrary = geraGrafo `suchThat` isValid
  220         where geraGrafo = do ns <- arbitrary
  221                              return $ Graph {nodes = fromList ns, edges = fromList (cEdges ns)}
  222 
  223 cEdges :: [v] -> [Edge v]
  224 cEdges [] = []
  225 cEdges [x] = [Edge x x]
  226 cEdges (h:t:z) = [Edge h t] ++ cEdges z
  227 
  228  
  229 prop_valid :: Graph Int -> Property
  230 prop_valid g = collect (length (edges g)) $ isValid g
  231 
  232 -- Gerador de DAGs
  233 dag :: (Ord v, Arbitrary v) => Gen (DAG v)
  234 dag = arbitrary `suchThat` isDAG
  235 
  236 prop_dag :: Property
  237 prop_dag = forAll (dag :: Gen (DAG Int)) $ \g -> collect (length (edges g)) $ isDAG g
  238 
  239 -- Gerador de florestas
  240 forest :: (Ord v, Arbitrary v) => Gen (Forest v)
  241 forest = arbitrary `suchThat` isForest
  242 
  243 prop_forest :: Property
  244 prop_forest = forAll (forest :: Gen (Forest Int)) $ \g -> collect (length (edges g)) $ isForest g

  253 -- Exemplo de uma propriedade QuickCheck para testar a função adj          
  254 prop_adj :: Graph Int -> Property
  255 prop_adj g = forAll (elements $ elems $ nodes g) $ \v -> adj g v `isSubsetOf` edges g
  256 
  257 -- Exemplos de propriedades QuickCheck para testar a função Swap
  258 prop_Swap :: Eq v => Edge v -> Bool
  259 prop_Swap  (Edge x y) = ((target(Edge x y)) == (source(swap(Edge x y)))) && ((source(Edge x y)) == (target(swap(Edge x y))))
  260 
  261 prop_swap2 :: Edge Int -> Bool
  262 prop_swap2 x = ((swap(swap x)) == x)
  263 
  264 -- Exemplos de propriedades QuickCheck para testar a função isEmpty
  265 prop_isEmpty :: Graph Int -> Bool
  266 prop_isEmpty x = (vazio x == isEmpty x)
  267 
  268 vazio :: Graph v -> Bool
  269 vazio x = Prelude.null (elems (nodes x))
  270 
  271 prop_isEmpty2 :: Graph Int -> Property
  272 prop_isEmpty2 x = isEmpty x ==> (length (nodes x)) == 0
  273 
  274 -- Exemplos de propriedades QuickCheck para testar a função isValid
  275 prop_isValid :: Graph Int -> Bool
  276 prop_isValid x = (Prelude.map source (toList(edges x)) `unionP` Prelude.map target (toList(edges x))) `isLista` (toList(nodes x)) == (isValid x)
  277 
  278 unionP :: Eq a => [a] -> [a] -> [a]
  279 unionP [] x = x
  280 unionP (h:t) (x:y) = if (x==h) then h: unionP t y
  281                      else h: unionP t (x:y)
  282 
  283 isLista :: [Int] -> [Int] -> Bool
  284 isLista [] [] = True
  285 isLista [] x = True
  286 isLista x [] = False
  287 isLista (x:y) (h:t) | (elem x (h:t)) = (isLista y (h:t))
  288                     | otherwise = False

  312 -- Exemplos de propriedades QuickCheck para testar a função union
  313 prop_union :: Graph Int -> Graph Int -> Bool
  314 prop_union g1 g2 = (prop_union_nodes g1 g2) && (prop_union_edges g1 g2)

  316 prop_union_nodes :: Graph Int -> Graph Int -> Bool
  317 prop_union_nodes x y = (fromList $ (elems (nodes x)) ++ (elems (nodes y))) == (fromList (elems (nodes (Graph.union x y))))
  318 
  319 prop_union_edges :: Graph Int -> Graph Int -> Bool
  320 prop_union_edges x y = (fromList $ (elems (edges x)) ++ (elems (edges y))) == (fromList (elems (edges (Graph.union x y))))
